Создать_ставку -> Событие: Выбрать событие
activate Создать_ставку
	activate Событие
		Событие -> Исход: Выбрать исход
	    activate Исход
    	Исход --> Создать_ставку: Исход
	    deactivate Исход
	deactivate Событие
	Создать_ставку ->  Пользователь: Проверка лимитов
 	   	activate Пользователь
 	       Пользователь --> Создать_ставку: Статус
   	 	deactivate Пользователь     
    Создать_ставку -> Ставка: Зарегистрировать ставку
 	    activate Ставка
		   	Ставка --> Создать_ставку: Ставка
 	    deactivate Ставка
deactivate Создать_ставку
